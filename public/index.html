    How to Message

    This step-by-step tutorial will help you configure and use messaging in your app.

    1. Create new App in the Messaging Console
    2. Set up GCM credentials in the Settings tab
    3. Import Android SDK into your mobile app
    4. Import MMX API Key into settings.xml of your mobile app
    5. Set up receipient mobile app to listen for messages 
      //Need to so some setup in the app's Application class
public class MyApplication extends Application {
  public void onCreate() {
    super.onCreate();
    MMXClient.registerWakeupListener(this, MyWakeupListener.class);
    MMXClient.setWakeupInterval(this, 60 * 1000); //if a poll wakeup is required
  }
}

public class MyWakeupListener implements MMXWakeupListener {
  private static final String TAG = MyWakeupListener.class.getSimpleName();
  public void onWakeupReceived(final Context applicationContext, Bundle extras) {
    //this gets called when a "wakeup" occurs, either from a GCM message or from the polling interval (if set)
    //the developer can either connect to their MMX or handle the the extras that came with the GCM intent
  }
}
